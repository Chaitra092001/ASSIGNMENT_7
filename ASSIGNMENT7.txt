1)SELECT NAME, CUSTOMER_NAME, S.CITY
FROM SALESMAN S, CUSTOMER C
WHERE S.SALESMAN_ID=C.SALESMAN_ID AND S.CITY=C.CITY;

2)SELECT ORDER_NO, PURCHASE_AMOUNT, CUSTOMER_NAME, C.CITY
FROM ORDERS O, CUSTOMER C
WHERE O.CUSTOMER_ID=C.CUSTOMER_ID AND PURCHASE_AMOUNT BETWEEN 500 AND 2000;

3)SELECT CUSTOMER_NAME, C.CITY, NAME AS SALESMAN_NAME,COMMISSION
FROM SALESMAN S, CUSTOMER C
WHERE S.SALESMAN_ID=C.SALESMAN_ID;

4)SELECT CUSTOMER_NAME, C.CITY , NAME ,COMMISSION
FROM SALESMAN S, CUSTOMER C
WHERE S.SALESMAN_ID=C.SALESMAN_ID AND S.COMMISSION > 0.12;

5)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, NAME AS SALESMAN_NAME, S.CITY AS SALESMAN_CITY, COMMISSION
FROM SALESMAN S, CUSTOMER C
WHERE S.SALESMAN_ID=C.SALESMAN_ID AND COMMISSION > 0.12 AND S.CITY!=C.CITY;

6)SELECT ORDER_NO, ORDER_DATE, PURCHASE_AMOUNT, CUSTOMER_NAME, GRADE, NAME AS SALESMAN_NAME, COMMISSION
FROM SALESMAN S, CUSTOMER C, ORDERS O
WHERE  S.SALESMAN_ID=C.SALESMAN_ID AND O.CUSTOMER_ID=C.CUSTOMER_ID;

7)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY,GRADE, NAME AS SALESMAN_NAME, S.CITY AS SALESMAN_CITY
FROM SALESMAN S, CUSTOMER C 
WHERE S.SALESMAN_ID=C.SALESMAN_ID 
ORDER BY CUSTOMER_ID ASC;

8)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY,GRADE, NAME AS SALESMAN_NAME, S.CITY AS SALESMAN_CITY
FROM SALESMAN S, CUSTOMER C 
WHERE S.SALESMAN_ID=C.SALESMAN_ID AND GRADE< 300
ORDER BY CUSTOMER_ID ASC;

9)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, ORDER_NO, ORDER_DATE, PURCHASE_AMOUNT
FROM  CUSTOMER C LEFT JOIN ORDERS O ON O.CUSTOMER_ID=C.CUSTOMER_ID
ORDER BY ORDER_DATE;

10)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, ORDER_NO, ORDER_DATE, PURCHASE_AMOUNT, NAME AS SALESMAN_NAME, COMMISSION
FROM 
 CUSTOMER C LEFT JOIN ORDERS O ON C.CUSTOMER_ID=O.CUSTOMER_ID
LEFT JOIN SALESMAN S ON O.SALESMAN_ID=S.SALESMAN_ID;

11)SELECT  CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, GRADE AS CUSTOMER_GRADE, NAME AS SALESMAN_NAME, S.CITY AS SALESMAN_CITY
FROM  SALESMAN S LEFT JOIN CUSTOMER C
ON S.SALESMAN_ID=C.SALESMAN_ID
ORDER BY SALESMAN_NAME;

12)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, ORDER_NO, ORDER_DATE, PURCHASE_AMOUNT
FROM CUSTOMER C LEFT JOIN ORDERS O
ON O.CUSTOMER_ID=C.CUSTOMER_ID;

13)SELECT CUSTOMER_NAME, C.CITY AS CUSTOMER_CITY, ORDER_NO, ORDER_DATE, PURCHASE_AMOUNT
FROM CUSTOMER C  JOIN ORDERS O
ON O.CUSTOMER_ID=C.CUSTOMER_ID WHERE GRADE IS NOT NULL;

14)SELECT * FROM SALESMAN CROSS JOIN CUSTOMER;

15)SELECT * FROM SALESMAN CROSS JOIN CUSTOMER WHERE SALESMAN.CITY=CUSTOMER.CITY;

16)SELECT * FROM SALESMAN CROSS JOIN CUSTOMER WHERE GRADE IS NOT NULL AND SALESMAN.CITY IS NOT NULL;

17)SELECT * FROM SALESMAN CROSS JOIN CUSTOMER 
WHERE GRADE IS NOT NULL AND SALESMAN.CITY != CUSTOMER.CITY AND SALESMAN.CITY IS NOT NULL ;
